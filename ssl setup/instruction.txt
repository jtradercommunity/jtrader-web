#1
sudo apt-get update
sudo apt-get install nginx

#2
sudo apt-get install letsencrypt

#3
letsencrypt certonly --standalone -d api.jtrader.com

#4
cd /etc/nginx/sites-available
cp default api.jtrader.com
nano api.jtrader.com

#5
upstream socket_nodes {
    ip_hash;
    server localhost:3000 weight=5;
}
server {
      index index.html index.htm;
      server_name api.jtrader.com;
      location / {
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_http_version 1.1;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header Host $host;
        proxy_pass http://socket_nodes;
      }
}
server {
        listen 443;
        server_name app.jtrader.com;
        index index.html index.htm;
        ssl on;
        ssl_certificate /etc/letsencrypt/live/api.jtrader.com/fullchain.pem;
        ssl_certificate_key /etc/letsencrypt/live/api.jtrader.com/privkey.pem;
        ssl_session_timeout 5m;
        ssl_protocols SSLv3 TLSv1 TLSv1.1 TLSv1.2;
        ssl_ciphers "HIGH:!aNULL:!MD5 or HIGH:!aNULL:!MD5:!3DES";
        ssl_prefer_server_ciphers on;
        location / {
                proxy_set_header Upgrade $http_upgrade;
                proxy_set_header Connection "upgrade";
                proxy_http_version 1.1;
                proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
                proxy_set_header Host $host;
                proxy_pass http://socket_nodes;
        }
}

#6
sudo ln -s /etc/nginx/sites-available/api.jtrader.com

#7
sudo service nginx restart

#renew
letsencrypt renew
